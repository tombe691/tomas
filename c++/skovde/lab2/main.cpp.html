16   // shuffle function where cards get new positions
17   void shuffledeck(string card[], int nrofcards)
18   {
19       // init random seed
20       srand(time(0));
21       //going through the positions
22       for (int i=0; i<nrofcards ;i++)
23       {
24           // create a random position of remaining ones
25           int randomposition = i + (rand() % (nrofcards -i));
26           //using standard method to swap to positions
27           swap(card[i], card[randomposition]);
28       }
29   }
30   //function to rank and compare colours
31   int getColourRank(char colour, string* colourname)
32   {
33       int rank = 0;
34       switch (colour) {
35           case 'S' :
36               rank = 4;
37               *colourname = "spades";
38               break;
39           case 'H' :
40               rank = 3;
41               *colourname = "hearts";
42               break;
43           case 'D' :
44               rank = 2;
45               *colourname = "diamonds";
46               break;
47           case 'C' :
48               rank = 1;
49               *colourname = "clubs";
50               break;
51           default :
52               cout << "Invalid colour" << endl;
53       }
54       return rank;
55   }

77   //function to compare card values, if equal call colour compare
78   int comparecards(string player1card, string player2card)
79   {
80       string player1colour = player1card.substr(0, 1);
81       string player1nr = player1card.substr(1);
82       char p1col = player1colour[0];
83       string player2colour = player2card.substr(0, 1);
84       string player2nr = player2card.substr(1);
85       char p2col = player2colour[0];
86       int winningplayer = 0;
87       int player1number = stoi(player1nr);
88       int player2number = stoi(player2nr);
89   
90       if (player1number > player2number){
91           winningplayer = 1;
92       }
93       else if (player1number < player2number){
94           winningplayer = 2;
95       }
96       else if (player1number == player2number){
97           winningplayer = comparecolours(p1col, p2col);
98       }
99       else {
100          cout << "should not happen";
101      }
102      return winningplayer;
103  }

154  void gameloop()
155  {
156      string symbols[] = {"S", "H", "D", "C"};
157      string b[52];
158      int count = 0;
159      for (int colour = 0; colour < 4;colour++) {
160          for (int nr=2; nr<15; nr++) {
161              b[(nr+count)-2] = symbols[colour]+to_string(nr);
162          }
163          count +=13;
164      }
165  //    printallcardsincurrentorder(b);
166      shuffledeck(b, 52);
167      string player1card;
168      string player2card;
169      string cardcolour;
170      int winningplayer = 0;
171      int roundnumber = 1;
172      int player1points = 0;
173      int player2points = 0;
174      int player1wins = 0;
175      int player1losses = 0;
176      int player2wins = 0;
177      int player2losses = 0;
178      for (int k=0; k<52 && roundnumber<4;) {
179          cout << "round: " << roundnumber << endl;
180          player1card = b[k];
181          getColourRank(player1card[0], &cardcolour);
182          //cout << "p1 colour = " << cardcolour << endl;
183          cout << "player one's card is ";
184          cout << printcorrectcardnames(player1card.substr(1)) << " of " << cardcolour << ", ";
185          k++;
186          player2card = b[k];
187          getColourRank(player2card[0], &cardcolour);
188          //cout << "p1 colour = " << cardcolour << endl;
189          cout << "player two's card is ";
190          cout << printcorrectcardnames(player2card.substr(1)) << " of " << cardcolour << "." << endl;
191          k++;
192          winningplayer = comparecards(player1card, player2card);
193          cout << "Winning player this round is: player " << winningplayer << endl;
194          if (winningplayer==1){
195              player1points+=1;
196              player1wins+=1;
197              player2losses+=1;
198          }
199          else if (winningplayer==2){
200              player2points+=1;
201              player2wins+=1;
202              player1losses+=1;
203          }
204          printcurrentstatus(player1points, player2points, player1wins,
205                             player1losses, player2wins, player2losses);
206          string next;
207          cin >> next;
208          cin.ignore();
209          roundnumber++;
210      }
211      printwinner(player1points, player2points);
212  }
